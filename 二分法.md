def lastPosition(self, nums, target):
        if not nums:
            return -1
        
        start, end = 0, len(nums) -1
        while start +1 < end:
            mid = (start + end) //2
            if nums[mid] < target:
                start = mid
            elif nums[mid] > target:
                end = mid
            else:
                start = mid

        if nums[end] == target:
            return end
        if nums[start] == target:
            return start


        return -1
